OPEN file
ENCODE file to BASE64 file_buffer
PACKETIZE file to packet_array
  FOR EACH byte in file_buffer
    PACK byte
    SEQ_NUM ++
  END FOR
  ADD EOT packet to packet_array

  TRANSMIT packet_array
    WHILE current_packet <= packet_array(last_packet) DO
      SEND packet(current_packet)
      WAITFORACK
        IF ACK received THEN
          last_ACK ++
          CONTINUE
        ELSE TIMEOUT(CONTINUE)
